#include<stdio.h>
int main()
{
   char str[100];
   int a;
   scanf("%d",&a);
   scanf("%[^\n]s",str);
   int key,i;
   scanf("%d",&key);
   for(i=0;str[i]!='\0';i++)
   {
      if(str[i]>=97 &&str[i]<=122)
      {
          str[i]=str[i]+key;
          //if(str[i]>122) //122 -> Z
            // str[i]=str[i]-26;
      }
    else if(str[i]>=65 &&str[i]<=90)
      {
          str[i]=str[i]+key;
          if(str[i]>90)
             str[i]=str[i]-26;
      }
   }
   printf("%s",str);
}
